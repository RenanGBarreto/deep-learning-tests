X, y = createDataSetL1Q5(1000, lookback=10, lookahead=1)
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=SEED, shuffle=True)

model = NeuralNetwork(10, [6, 1], isClassification=False, lr=0.5, autoEncode=False, activation='tanh',
                      activation_last_layer='tanh', seed=SEED, momentum=0.2)

print("TRAINING DETAILS:")
model.fit(X=X_train, y=y_train, epochs=200, verbose=False, plot=True, shuffle=True, batch_size=1, lr_decay=0.95)

y_test_true = np.reshape(y_test, (len(y_test)))
y_test_pred = np.reshape(model.predict(X_test), (len(X_test)))
y_train_true = np.reshape(y_train, (len(y_train)))
y_train_pred = np.reshape(model.predict(X_train), (len(X_train)))

print("MSE TRAINING DATASET:\n", mean_absolute_error(y_train_true, y_train_pred), "\n")
print("MSE TEST DATASET:\n", mean_absolute_error(y_test_true, y_test_pred), "\n")

print("PREDICTION CHART:")

X_sequence, y_sequence = createDataSetL1Q5(850, lookback=10, lookahead=1)
y_sequence_true = np.reshape(y_sequence, (len(y_sequence)))
y_sequence_pred = np.reshape(model.predict(X_sequence), (len(y_sequence)))

fig, ax = plt.subplots(figsize=(10,7))
ax.plot(np.linspace(0, 50, len(y_sequence_true)), y_sequence_true, label="True Value")
ax.plot(np.linspace(0, 50, len(y_sequence_true)), y_sequence_pred, label="Predicted Value")
ax.set(xlabel='', ylabel='Function Value', title='True x Predicted: 1 steps prediction')
ax.legend()
ax.grid()
plt.show()

print("NETWORK ARCHITECTURE:")
model.draw(size=(10,8))
