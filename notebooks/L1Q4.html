model = NeuralNetwork(2, [8, 8], isClassification=True, lr=0.05, autoEncode=True, activation='tanh',
                      activation_last_layer='tanh', seed=SEED)

print("TRAINING DETAILS:")
model.fit(X=X_train, y=y_train, epochs=9, verbose=False, plot=True, shuffle=True)

print("NETWORK STATE AFTER TRAINING:\n", model, "\n")
y_test_true = np.reshape(y_test, (len(y_test)))
y_test_pred = np.reshape(model.predict(X_test), (len(X_test)))

print("CLASSIFICATION REPORT:\n", classification_report(y_test_true, y_test_pred), "\n")

import matplotlib.pyplot as plt

y_test_true = np.reshape(y_test, (len(y_test)))
y_test_pred = np.reshape(model.predict(X_test), (len(X_test)))

groups = []
for c in range(8):
    groups.append([[],[]])

for i in range(len(X_test)):
    groups[y_test_pred[i]-1][0].append(X_test[i][0])
    groups[y_test_pred[i]-1][1].append(X_test[i][1])

# Plot
fig, ax = plt.subplots(figsize=(10,7))
ax.margins(0.05)
ax.set(title='Class Prediction on Training Set')
for c in range(len(groups)): 
    ax.plot(groups[c][0], groups[c][1], marker='o', linestyle='', ms=8, label="C"+str(c+1))
ax.legend()
plt.show()
